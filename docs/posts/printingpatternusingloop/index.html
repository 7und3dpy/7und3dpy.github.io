#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int arr[10000][10000];
int n;

void init()
{
    for (int i = 0; i < 2*n-1;i++)
    {
        for (int j = 0; j < 2*n-1;j++)
        {
            arr[i][j] = -1;
        }
    }
}

void changerow(int row)
{
    for (int c = row;c < (2*n - 1 - row);c++)
    {
        if (arr[row][c] == -1)
        {
            arr[row][c] = n - row;
            arr[2*n - 1 - row - 1][c] = n - row;
        }
    }
}

void changecol(int col)
{
    for (int r = col;r < (2*n -1 - col);r++)
    {
        if (arr[r][col] == -1)
        {
            arr[r][col] = n - col;
            arr[r][2*n - 1 - col - 1] = n - col;       
        }
    }   
}

void solution()
{
    for (int i = 0;i< 2*n -1;i++)
    {
        for (int j = 0; j < 2*n - 1;j++)
        {
            printf("%d ", arr[i][j]);
        }
        printf("\n");
    }
}
int main() 
{
    scanf("%d", &n);
  	// Complete the code to print the pattern.
    init();
    for (int i = 0; i < n;i++)
    {
        changerow(i);
        changecol(i);
    }
    solution();
    return 0;
}
